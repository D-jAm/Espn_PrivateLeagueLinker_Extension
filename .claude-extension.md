# How We Lookin - ESPN Cookie Helper Chrome Extension

## Project Overview

This Chrome extension helps users extract their ESPN Fantasy Football authentication cookies (ESPN_S2 and SWID) to sync their private leagues with the "How We Lookin" web application for playoff odds calculation.

**Purpose**: Enable users to access their own private ESPN fantasy leagues without manually finding cookies in browser DevTools.

**Main Web App Repository**: `c:\Users\dylan\CodingProjects\AMiFucked\`

## Technical Stack

- **Manifest Version**: V3 (current Chrome standard)
- **Language**: Vanilla JavaScript (no build tools required)
- **Chrome APIs Used**:
  - `chrome.cookies` - Read ESPN cookies
  - `chrome.storage` - Optional: cache extracted cookies
- **Permissions Required**:
  - `cookies` - To access ESPN cookies
  - `host_permissions` for `https://*.espn.com/*`

### popup/popup.js

**Requirements**:
- Extract cookies using `chrome.cookies.get()`
- Handle cases where user is not logged in
- Copy to clipboard functionality
- Show/hide different states (loading, success, error)
- Truncate long cookie values for display
- Error handling with user-friendly messages

**Implementation**:
```javascript
// State management
let extractedCookies = {
  espn_s2: null,
  swid: null
};

// DOM elements
const extractBtn = document.getElementById('extractBtn');
const results = document.getElementById('results');
const successState = document.getElementById('successState');
const errorState = document.getElementById('errorState');
const loadingState = document.getElementById('loadingState');
const espnS2ValueEl = document.getElementById('espnS2Value');
const swidValueEl = document.getElementById('swidValue');
const copyS2Btn = document.getElementById('copyS2');
const copySWIDBtn = document.getElementById('copySWID');

// Extract cookies function
async function extractCookies() {
  try {
    // Show loading state
    showState('loading');

    // Get ESPN_S2 cookie
    const espnS2Cookie = await chrome.cookies.get({
      url: 'https://fantasy.espn.com',
      name: 'espn_s2'
    });

    // Get SWID cookie
    const swidCookie = await chrome.cookies.get({
      url: 'https://fantasy.espn.com',
      name: 'SWID'
    });

    // Check if both cookies exist
    if (!espnS2Cookie || !swidCookie) {
      showState('error');
      console.error('Cookies not found. User may not be logged in.');
      return;
    }

    // Store extracted cookies
    extractedCookies.espn_s2 = espnS2Cookie.value;
    extractedCookies.swid = swidCookie.value;

    // Display cookies (truncated)
    espnS2ValueEl.textContent = truncateCookie(espnS2Cookie.value);
    espnS2ValueEl.title = espnS2Cookie.value; // Full value in tooltip
    swidValueEl.textContent = swidCookie.value;
    swidValueEl.title = swidCookie.value;

    // Show success state
    showState('success');

    console.log('‚úÖ Cookies extracted successfully');

  } catch (error) {
    console.error('Error extracting cookies:', error);
    showState('error');
  }
}

// Show different states
function showState(state) {
  results.classList.remove('hidden');
  successState.classList.add('hidden');
  errorState.classList.add('hidden');
  loadingState.classList.add('hidden');

  if (state === 'success') {
    successState.classList.remove('hidden');
  } else if (state === 'error') {
    errorState.classList.remove('hidden');
  } else if (state === 'loading') {
    loadingState.classList.remove('hidden');
  }
}

// Truncate long cookie values for display
function truncateCookie(value, maxLength = 50) {
  if (value.length <= maxLength) return value;
  return value.substring(0, maxLength) + '...';
}

// Copy to clipboard function
async function copyToClipboard(text, buttonEl) {
  try {
    await navigator.clipboard.writeText(text);

    // Visual feedback
    const originalText = buttonEl.textContent;
    buttonEl.textContent = '‚úì Copied!';
    buttonEl.classList.add('copied');

    setTimeout(() => {
      buttonEl.textContent = originalText;
      buttonEl.classList.remove('copied');
    }, 2000);

    console.log('‚úÖ Copied to clipboard');
  } catch (error) {
    console.error('Failed to copy:', error);
    alert('Failed to copy. Please try again.');
  }
}

// Event listeners
extractBtn.addEventListener('click', extractCookies);

copyS2Btn.addEventListener('click', () => {
  copyToClipboard(extractedCookies.espn_s2, copyS2Btn);
});

copySWIDBtn.addEventListener('click', () => {
  copyToClipboard(extractedCookies.swid, copySWIDBtn);
});

// Optional: Auto-extract on popup open (remove if you want manual trigger only)
// document.addEventListener('DOMContentLoaded', extractCookies);
```
